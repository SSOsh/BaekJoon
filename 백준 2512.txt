#include <iostream>
using namespace std;

int arr[10000];

int main() {
	int n, m;
	int result;
	int mid = 0, low = 0, high = 0;
	scanf("%d", &n);
	for (int i = 0; i < n; i++) {
		scanf("%d", &arr[i]);
		if (high < arr[i]) high = arr[i];
	}
	scanf("%d", &m);

	while (1) {
		result = 0;
		mid = (low + high) / 2;
		if (mid < low || mid > high) {
			break;
		}
		for (int i = 0; i < n; i++) {
			if (arr[i] > mid) {
				result += mid;
			}
			else {
				result += arr[i];
			}
		}
		if (m < result) {
			high = mid - 1;
		}
		else if(m > result) {
			low = mid + 1;
		}
		else {
			break;
		}
	}

	printf("%d", mid);
	return 0;
}
