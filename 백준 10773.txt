#include <iostream>
#include <stack>
using namespace std;

stack<int> si;

int main() {
	int n, m, result = 0;
	scanf("%d", &n);
	for (int i = 0; i < n; i++) {
		scanf("%d", &m);
		if (m != 0) {
			si.push(m);
		}
		else {
			si.pop();
		}
	}
	while (!si.empty()) {
		result += si.top();
		si.pop();
	}
	printf("%d", result);

	return 0;
}
